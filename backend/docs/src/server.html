132def decode_token(token: str) -> dict:
133    try:
134        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
135        return payload
136    except jwt.ExpiredSignatureError:
137        raise HTTPException(
138            status_code=status.HTTP_401_UNAUTHORIZED, detail="Token has expired"
139        )
140    except (
141        jwt.InvalidTokenError
142    ):  # Changed from jwt.JWTError to jwt.InvalidTokenError for more specific handling
143        raise HTTPException(
144            status_code=status.HTTP_401_UNAUTHORIZED,
145            detail="Could not validate credentials",
146        )

153def get_current_user(
154    credentials: HTTPAuthorizationCredentials = Depends(security),
155    db: Session = Depends(get_db),
156) -> User:
157    token = credentials.credentials
158    payload = decode_token(token)
159    user_id_str: str = payload.get("sub")
160
161    if user_id_str is None:
162        raise HTTPException(
163            status_code=status.HTTP_401_UNAUTHORIZED,
164            detail="Could not validate credentials",
165        )
166
167    try:
168        user_id = int(user_id_str)
169    except (ValueError, TypeError):
170        raise HTTPException(
171            status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid token format"
172        )
173
174    user = db.query(User).filter(User.id == user_id).first()
175    if user is None:
176        raise HTTPException(
177            status_code=status.HTTP_401_UNAUTHORIZED, detail="User not found"
178        )
179
180    return user

214@app.post("/api/auth/login", response_model=Token)
215def login(user_data: UserLogin, db: Session = Depends(get_db)):
216    # Find user by email
217    user = get_user_by_email(db, user_data.email)
218
219    if not user or not verify_password(user_data.password, user.hashed_password):
220        raise HTTPException(
221            status_code=status.HTTP_401_UNAUTHORIZED,
222            detail="Incorrect email or password",
223        )
224
225    # Create access token
226    access_token_expires = timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
227    access_token = create_access_token(
228        data={"sub": str(user.id)}, expires_delta=access_token_expires
229    )

329@app.get("/api/public/models", response_model=List[Model3DResponse])
330def get_public_models(
331    search: Optional[str] = None,
332    skip: int = 0,
333    limit: int = 100,
334    db: Session = Depends(get_db),
335):
336    query = db.query(Model3D).join(User).filter(Model3D.is_public == True)

347    return models

353async def generate_model_endpoint(
354    file: UploadFile = File(...),
355    current_user: User = Depends(get_current_user),
356    db: Session = Depends(get_db),
357):
358    # Save uploaded file
359    file_ext = os.path.splitext(file.filename)[1]
360    if not file_ext:
361        file_ext = ".png"  # Default to png if no extension
362
363    file_name = f"{uuid.uuid4()}{file_ext}"
364    file_path = UPLOAD_DIR / file_name
365
366    with open(file_path, "wb") as buffer:
367        shutil.copyfileobj(file.file, buffer)
368
369    # Generate 3D model
370    try:
371        model_url = await generate_3d(str(file_path))
372    except Exception as e:
373        # Cleanup uploaded file on failure
374        if os.path.exists(file_path):
375            os.remove(file_path)
376        raise HTTPException(status_code=500, detail=str(e))
377
378    # Create DB record
379    return create_model(
380        db,
381        {
382            "reference_image_path": str(file_path),
383            "generated_3d_path": model_url,
384        },
385        current_user.id,
386    )

389@app.post("/api/generate-2d")
390async def generate_2d_endpoint(
391    instructions: str = Form(...),
392    file: Optional[UploadFile] = File(None),
393    current_user: User = Depends(get_current_user),
394):
395    base_image_b64 = None
396    if file:
397        file_content = await file.read()
398        base_image_b64 = base64.b64encode(file_content).decode("utf-8")
399
400    try:
401        # Generate 2D image
402        file_path = await generate_2d(
403            prompt=instructions,
404            base_image_b64=base_image_b64,
405            output_dir=str(UPLOAD_2D_DIR),
406        )
407
408        # Return URL relative to server
409        # Assuming we mount uploads dir or similar.
410        # Let's mount uploads dir to serve static files.
411        relative_path = os.path.relpath(file_path, ".")
412        return {"image_url": f"/{relative_path}"}
413
414    except ClientError as e:
415        print(f"Gemini API Error: {e}")
416        raise HTTPException(status_code=e.code, detail=e.message)
417    except Exception as e:
418        import traceback
419
420        traceback.print_exc()
421        print(f"Error generating 2D image: {e}")
422        raise HTTPException(status_code=500, detail=str(e))
